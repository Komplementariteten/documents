 	\int{\frac{1}{\sqrt{1 + x^2}}} dx =

 	\end{cases}

 	\end{cases}

 	\arcoth{x} + C_2 = \frac{1}{2} \ln{\frac{x - 1} {x - 1}} + C_2 & \text{$|x| > 1$}
